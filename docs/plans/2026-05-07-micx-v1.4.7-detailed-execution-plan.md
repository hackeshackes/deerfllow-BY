# v1.4.7 详细开发计划：智能演示生成 (ppt-master)

## 目标

用 ppt-master 完全替换现有的 PPT 生成方式，实现真正的可编辑 PPTX 输出。

## 架构概览

```
┌─────────────────────────────────────────────────────────────┐
│                    Frontend (Next.js)                       │
│  用户输入主题 → 调用后端 API → 下载生成的 PPTX 文件           │
└─────────────────────────┬───────────────────────────────────┘
                          │ POST /api/ppt/generate
                          ▼
┌─────────────────────────────────────────────────────────────┐
│                  Backend Gateway (FastAPI)                   │
│  /api/ppt/generate - 接收主题，返回 PPTX 文件流               │
└─────────────────────────┬───────────────────────────────────┘
                          │
                          ▼
┌─────────────────────────────────────────────────────────────┐
│              PPT Generation Service (Python)                 │
│  - 调用 LLM 生成 PPT 结构 (slides, content, layout)          │
│  - 调用 ppt-master 生成 SVG                                  │
│  - 转换为原生 PPTX (DrawingML shapes)                        │
│  - 返回文件路径                                              │
└─────────────────────────────────────────────────────────────┘
```

```python
"""
PPT Generation Service - 基于 ppt-master
"""
import os
import uuid
import json
from pathlib import Path
from typing import Optional

class PPTService:
    """PPT 生成服务"""

    def __init__(self, workspace_dir: str = "/tmp/ppt-workspaces"):
        self.workspace_dir = Path(workspace_dir)
        self.workspace_dir.mkdir(parents=True, exist_ok=True)

    async def generate_ppt(
        self,
        topic: str,
        num_slides: int = 8,
        style: str = "gradient-modern",
        aspect_ratio: str = "16:9",
    ) -> str:
        """
        生成 PPT

        Args:
            topic: PPT 主题
            num_slides: 幻灯片数量 (默认 8)
            style: 风格 (gradient-modern, dark-premium, glassmorphism 等)
            aspect_ratio: 比例 (16:9 或 4:3)

        Returns:
            生成的 PPTX 文件路径
        """
        # 1. 创建工作空间
        session_id = str(uuid.uuid4())
        session_dir = self.workspace_dir / session_id
        session_dir.mkdir(parents=True, exist_ok=True)

        # 2. 准备项目目录结构
        projects_dir = session_dir / "projects" / session_id
        projects_dir.mkdir(parents=True, exist_ok=True)

        # 3. 创建源文件 (topic.txt)
        source_file = projects_dir / "sources" / "topic.txt"
        source_file.parent.mkdir(parents=True, exist_ok=True)
        source_file.write_text(f"主题: {topic}\n风格: {style}\n幻灯片数量: {num_slides}", encoding="utf-8")

        # 4. 调用 ppt-master 生成 PPT
        # 这里需要调用 ppt-master 的核心逻辑
        pptx_path = await self._call_ppt_master(
            session_dir=session_dir,
            session_id=session_id,
            topic=topic,
            num_slides=num_slides,
            style=style,
            aspect_ratio=aspect_ratio,
        )

        return pptx_path

    async def _call_ppt_master(
        self,

        session_id: str,
        topic: str,

        aspect_ratio: str,
```

#### 2.2 创建 API Router

**文件**: `backend/app/gateway/routers/ppt.py`

```python
"""
PPT Generation API Router
"""
from fastapi import APIRouter, HTTPException, BackgroundTasks
from fastapi.responses import FileResponse, StreamingResponse
from pydantic import BaseModel
from typing import Optional
import os

from app.services.ppt_service import PPTService

router = APIRouter(prefix="/api/ppt", tags=["PPT Generation"])

# 全局服务实例
_ppt_service: Optional[PPTService] = None

def get_ppt_service() -> PPTService:
    global _ppt_service
    if _ppt_service is None:
        _ppt_service = PPTService()
    return _ppt_service


class PPTGenerateRequest(BaseModel):
    """PPT 生成请求"""
    topic: str  # PPT 主题
    num_slides: int = 8  # 幻灯片数量
    style: str = "gradient-modern"  # 风格
    aspect_ratio: str = "16:9"  # 比例


class PPTGenerateResponse(BaseModel):
    """PPT 生成响应"""
    task_id: str  # 任务 ID，用于查询进度/下载
    status: str  # pending / processing / completed / failed
    message: str


@router.post("/generate", response_model=PPTGenerateResponse)
async def generate_ppt(request: PPTGenerateRequest):
    """
    生成 PPT

    用户输入主题，后端调用 ppt-master 生成可编辑的 PPTX 文件
    """
    service = get_ppt_service()

    try:
        # 调用服务生成 PPT
        pptx_path = await service.generate_ppt(
            topic=request.topic,
            num_slides=request.num_slides,
            style=request.style,
            aspect_ratio=request.aspect_ratio,
        )

        return PPTGenerateResponse(
            task_id=os.path.basename(pptx_path).replace(".pptx", ""),
            status="completed",
            message="PPT 生成成功"
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"PPT 生成失败: {str(e)}")


@router.get("/download/{task_id}")
async def download_ppt(task_id: str):
    """
    下载生成的 PPT 文件
    """
    service = get_ppt_service()

    pptx_path = service.get_output_path(task_id)

    if not os.path.exists(pptx_path):
        raise HTTPException(status_code=404, detail="PPT 文件不存在或已过期")

    return FileResponse(
        path=pptx_path,
        filename=f"presentation_{task_id}.pptx",
        media_type="application/vnd.openxmlformats-officedocument.presentationml.presentation"
    )
```

### Phase 3: 前端实现

#### 3.1 创建 PPT 生成 UI 组件

**文件**: `frontend/src/components/workspace/ppt/ppt-generator.tsx`

```tsx
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";

const PPT_STYLES = [
  { value: "gradient-modern", label: "现代渐变", description: "适合创业公司、科技产品" },
  { value: "dark-premium", label: "深色高端", description: "适合高管汇报、奢华品牌" },
  { value: "glassmorphism", label: "玻璃拟态", description: "适合 AI/SaaS 产品演示" },
  { value: "keynote", label: "Keynote 风格", description: "Apple 主题演示" },
  { value: "minimal-swiss", label: "瑞士极简", description: "适合架构、设计演示" },
];

export function PPTGenerator() {
  const [topic, setTopic] = useState("");
  const [numSlides, setNumSlides] = useState(8);
  const [style, setStyle] = useState("gradient-modern");
  const [isGenerating, setIsGenerating] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleGenerate = async () => {
    if (!topic.trim()) {
      setError("请输入 PPT 主题");
      return;
    }

    setIsGenerating(true);
    setError(null);

    try {
      const response = await fetch("/api/ppt/generate", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ topic, num_slides: numSlides, style }),
      });

      if (!response.ok) {
        throw new Error("生成失败");
      }

      const data = await response.json();

      // 下载生成的 PPT
      window.location.href = `/api/ppt/download/${data.task_id}`;
    } catch (err) {
      setError(err instanceof Error ? err.message : "生成失败");
    } finally {
      setIsGenerating(false);
    }
  };

  return (
    <div className="p-6 max-w-2xl mx-auto">
      <h2 className="text-2xl font-bold mb-6">生成演示文稿</h2>

      <div className="space-y-4">
        {/* 主题输入 */}
        <div>
          <label className="block text-sm font-medium mb-2">PPT 主题</label>
          <Input
            value={topic}
            onChange={(e) => setTopic(e.target.value)}
            placeholder="例如：2024年Q3产品迭代总结"
            disabled={isGenerating}
          />
        </div>

        {/* 幻灯片数量 */}
        <div>
          <label className="block text-sm font-medium mb-2">幻灯片数量</label>
          <Input
            type="number"
            min={3}
            max={20}
            value={numSlides}
            onChange={(e) => setNumSlides(parseInt(e.target.value) || 8)}
            disabled={isGenerating}
          />
        </div>

        {/* 风格选择 */}
        <div>
          <label className="block text-sm font-medium mb-2">设计风格</label>
          <Select value={style} onValueChange={setStyle} disabled={isGenerating}>
            <SelectTrigger>
              <SelectValue />
            </SelectTrigger>
            <SelectContent>
              {PPT_STYLES.map((s) => (
                <SelectItem key={s.value} value={s.value}>
                  {s.label} - {s.description}
                </SelectItem>
              ))}
            </SelectContent>
          </Select>
        </div>

        {/* 错误提示 */}
        {error && (
          <div className="text-red-500 text-sm">{error}</div>
        )}

        {/* 生成按钮 */}
        <Button
          onClick={handleGenerate}
          disabled={isGenerating || !topic.trim()}
          className="w-full"
        >
          {isGenerating ? "生成中..." : "生成 PPT"}
        </Button>
      </div>
    </div>
  );
}
```

### Phase 4: ppt-master 集成核心实现

#### 4.1 分析 ppt-master 核心逻辑

ppt-master 的核心流程：
1. **内容提取**: 从文档/文本提取关键信息
2. **结构规划**: 确定幻灯片结构 (标题、内容、总结)
3. **SVG 生成**: 调用 LLM 生成每页 SVG
4. **PPTX 转换**: 将 SVG 转换为原生 DrawingML shapes

关键文件：
- `skills/ppt-master/SKILL.md` - 核心工作流
- `skills/ppt-master/scripts/create_pptx.py` - PPTX 创建脚本
- `skills/ppt-master/scripts/svg_to_pptx.py` - SVG → PPTX 转换

#### 4.2 集成策略

由于 ppt-master 设计为在 AI IDE 中运行，我们需要提取其核心逻辑并作为服务调用：

**方案**: 创建 Python 包装脚本，暴露为 CLI/API

```python
# backend/services/ppt-master-integration/cli.py

"""
PPT Master 集成 CLI
提供简单的命令行接口供后端调用
"""
import argparse
import asyncio
import json
import sys
from pathlib import Path

# 添加 ppt-master 到路径
ppt_master_path = Path(__file__).parent / "ppt-master"
sys.path.insert(0, str(ppt_master_path))

from skills.ppt_master.scripts.create_pptx import create_pptx_from_content


async def main():
    parser = argparse.ArgumentParser(description="PPT Master CLI")
    parser.add_argument("--topic", required=True, help="PPT 主题")
    parser.add_argument("--num-slides", type=int, default=8, help="幻灯片数量")
    parser.add_argument("--style", default="gradient-modern", help="设计风格")
    parser.add_argument("--output", required=True, help="输出路径")
    parser.add_argument("--api-key", help="AI API Key (可选)")
    parser.add_argument("--model", default="claude", help="AI 模型")

    args = parser.parse_args()

    try:
        # 生成 PPT
        result = await create_pptx_from_content(
            topic=args.topic,
            num_slides=args.num_slides,
            style=args.style,
            output_path=args.output,
            api_key=args.api_key,
            model=args.model,
        )

        print(json.dumps({
            "success": True,
            "output": result,
        }))
    except Exception as e:
        print(json.dumps({
            "success": False,
            "error": str(e),
        }))
        sys.exit(1)


if __name__ == "__main__":
    asyncio.run(main())
```
