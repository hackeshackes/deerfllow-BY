### 3.2 搜索实现

```python
# 伪代码
def search_threads(query: str | None, tag_ids: list[int], workspace_id: str | None, ...):
    q = session.query(Thread)
    if query:
        # 标题匹配
        q = q.filter(Thread.title.ilike(f"%{query}%"))
        # 消息内容匹配（子查询）
        subq = session.query(Message.thread_id).filter(Message.content.ilike(f"%{query}%"))
        q = q.filter(Thread.id.in_(subq))
    if tag_ids:
        q = q.join(ThreadTag).filter(ThreadTag.tag_id.in_(tag_ids))
    if workspace_id:
        q = q.filter(Thread.workspace_id == workspace_id)
    ...
```
