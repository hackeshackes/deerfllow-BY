```python
@router.delete("/skills/{name}")
async def delete_skill(name: str, request: Request):
    user = require_user(request)
    skill = get_skill(name)

    # 权限校验
    if skill.source == "system":
        raise HTTPException(403, "系统内置技能无法删除")

    if skill.source == "admin" and not user.is_owner:
        raise HTTPException(403, "管理员技能仅管理员可删除")

    if skill.source == "user" and skill.installed_by != user.id and not user.is_owner:
        raise HTTPException(403, "只能删除自己安装的技能")

    # 执行删除
    ...
```

2. 删除按钮权限控制

**前端权限逻辑**:
```typescript
function canDeleteSkill(skill: SkillResponse, currentUser: User): boolean {
  if (skill.source === 'system') return false;
  if (skill.source === 'admin' && !currentUser.isOwner) return false;
  if (skill.source === 'user' && skill.installedBy !== currentUser.id && !currentUser.isOwner) return false;
  return true;
}
```
