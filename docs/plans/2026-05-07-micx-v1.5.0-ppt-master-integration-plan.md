## 6. 核心模块实现

### 6.1 PPTMasterService

```python
# backend/app/services/ppt_master/service.py

import asyncio
import subprocess
from pathlib import Path
from typing import Optional
from app.services.ppt_master.project_manager import ProjectManager
from app.services.ppt_master.converter import SourceConverter
from app.services.ppt_master.executor import Executor
from app.services.ppt_master.quality_check import QualityChecker
from app.services.ppt_master.post_processor import PostProcessor
from app.services.ppt_master.svg_to_pptx import SVGToPPTX

class PPTMasterService:
    """
    PPT Master 集成服务
    封装完整的 AI 工作流程: 源文档 → Markdown → SVG → PPTX
    """

    def __init__(self, workspace_dir: str = "/tmp/ppt-projects"):
        self.workspace_dir = Path(workspace_dir)
        self.project_manager = ProjectManager(self.workspace_dir)
        self.converter = SourceConverter()
        self.quality_checker = QualityChecker()

    async def generate_from_topic(
        self,
        topic: str,
        template: Optional[str] = None,
        format: str = "ppt169",
        pages: int = 10,
        **kwargs
    ) -> dict:
        """
        从主题生成 PPT (AI 模式)

        流程:
        1. 创建项目
        2. 源文档转换 (如果是文档输入)
        3. Strategist 设计规格确认
        4. Executor 生成 SVG
        5. Quality Check
        6. Post-processing → PPTX
        """
        # 1. 创建项目
        project = await self.project_manager.create(
            name=topic,
            format=format
        )

        # 2. 转换源文档 (如果是文档)
        if kwargs.get("source_file"):
            await self.converter.convert(kwargs["source_file"], project.path)

        # 3. 生成设计规格 (使用 AI)
        spec = await self._generate_design_spec(topic, template, **kwargs)

        # 4. Executor 生成 SVG
        svg_output = project.path / "svg_output"
        await self.executor.generate(spec, svg_output, pages=pages)

        # 5. Quality Check
        await self.quality_checker.check(svg_output)

        # 6. Post-processing → PPTX
        post_proc = PostProcessor(project.path)
        final_pptx = await post_proc.process()

        return {
            "success": True,
            "project_id": project.id,
            "pptx_path": str(final_pptx),
            "download_url": f"/api/ppt/download/{project.id}"
        }

    async def generate_from_document(
        self,
        file_path: str,
        template: Optional[str] = None,
        **kwargs
    ) -> dict:
        """从文档 (PDF/DOCX/URL) 生成 PPT"""
        project = await self.project_manager.create_from_source(file_path)

        # 转换 → 设计规格 → 生成 → 后处理
        ...

    async def extract_template(
        self,
        pptx_path: str,
        template_id: str,
        **kwargs
    ) -> dict:
        """从现有 PPTX 提取模板"""
        extractor = TemplateExtractor()
        template_path = await extractor.extract(pptx_path, template_id)

        return {
            "success": True,
            "template_id": template_id,
            "template_path": str(template_path)
        }
```

### 6.2 SVG to PPTX 转换器

```python
# backend/app/services/ppt_master/svg_to_pptx.py

import subprocess
from pathlib import Path
from typing import Optional, Literal

class SVGToPPTXConverter:
    """
    SVG → PPTX 转换核心
    基于 ppt-master svg_to_pptx.py
    """

    def __init__(self, skill_dir: Path):
        self.skill_dir = skill_dir
        self.script = skill_dir / "scripts" / "svg_to_pptx.py"

    async def convert(
        self,
        project_path: Path,
        stage: Literal["final", "draft"] = "final",
        transition: str = "fade",
        transition_duration: float = 0.5,
        office_compat: bool = True,
        output_format: Literal["native", "svg", "both"] = "native"
    ) -> dict:
        """
        执行 SVG → PPTX 转换

        Args:
            project_path: 项目路径
            stage: final/draft
            transition: 转场效果 (fade/slide/none)
            transition_duration: 转场持续时间 (秒)
            office_compat: Office 兼容模式 (PNG fallback)
            output_format: 输出格式 (native/svg/both)
        """
        cmd = [
            "python3", str(self.script),
            str(project_path),
            "-s", stage,
            "-t", transition,
            "-d", str(transition_duration),
        ]

        if office_compat:
            cmd.append("--office-compat")
        if output_format == "native":
            cmd.append("--only-native")
        elif output_format == "svg":
            cmd.append("--only-legacy")

        result = subprocess.run(cmd, capture_output=True, text=True)

        if result.returncode != 0:
            raise ConversionError(f"SVG→PPTX 失败: {result.stderr}")

        return {
            "success": True,
            "output": project_path / "exports"
        }
```

### 6.3 模板提取器

```python
# backend/app/services/ppt_master/template_extractor.py

from pathlib import Path
import zipfile
from lxml import etree
from typing import dict, Optional

class TemplateExtractor:
    """
    从现有 PPTX 提取主题/模板
    基于 ppt-master pptx_template_import.py
    """

    THEME_NAMESPACES = {
        'a': 'http://schemas.openxmlformats.org/drawingml/2006/main',
        'r': 'http://schemas.openxmlformats.org/officeDocument/2006/relationships',
    }

    async def extract(self, pptx_path: Path, template_id: str) -> dict:
        """提取 PPTX 中的主题元素"""
        with zipfile.ZipFile(pptx_path, 'r') as z:
            # 读取主题
            theme_xml = self._read_theme(z)
            theme_colors = self._parse_theme_colors(theme_xml)
            theme_fonts = self._parse_theme_fonts(theme_xml)

            # 读取母版
            slide_masters = self._read_slide_masters(z)

            # 提取背景/装饰
            backgrounds = self._extract_backgrounds(z)

        return {
            "template_id": template_id,
            "theme_colors": theme_colors,
            "theme_fonts": theme_fonts,
            "slide_layouts": slide_masters,
            "backgrounds": backgrounds
        }

    def _parse_theme_colors(self, theme_xml) -> dict:
        """解析主题颜色"""
        colors = {}
        color_el = theme_xml.find('.//a:dk1', self.THEME_NAMESPACES)
        if color_el is not None:
            colors['dark1'] = color_el.get('lastClr')

        # ... 更多颜色解析
        return colors
```

## 12. 风险与注意事项

### 12.1 AI 模型依赖

PPT Master 依赖 AI 模型生成内容和设计，需要配置有效的 API Key。建议:
- 使用 OpenAI GPT-4o (默认)
- 支持 Gemini 2.5 (成本优化)
- 支持 Claude (质量优先)

### 12.2 网络问题

ppt-master 仓库可能需要通过代理克隆或直接下载 ZIP。

### 12.3 系统依赖

Cairo 和 Pandoc 是可选但推荐的依赖，建议在 Dockerfile 中预装。

### 12.4 权限问题

Docker 容器内需要写入 `/workspace/ppt-projects` 目录。
