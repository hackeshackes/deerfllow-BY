```python
"""Tests for the token-refresh helper used by IM connectors."""
import time

import pytest

from app.gateway.connectors.token_refresh import CachedToken, TokenRefresher


class FakeFetcher:
    def __init__(self, tokens):
        self.tokens = list(tokens)
        self.calls = 0

    async def __call__(self):
        self.calls += 1
        return self.tokens.pop(0)


@pytest.mark.asyncio
async def test_cached_token_returns_initial_value_within_ttl():
    fetcher = FakeFetcher(["tok-1"])
    t = CachedToken(fetcher=fetcher, ttl_seconds=7200)
    assert await t.get() == "tok-1"
    assert fetcher.calls == 1


@pytest.mark.asyncio
async def test_cached_token_refreshes_after_ttl():
    fetcher = FakeFetcher(["tok-1", "tok-2"])
    t = CachedToken(fetcher=fetcher, ttl_seconds=0)
    # ttl=0 → always expired
    assert await t.get() == "tok-1"
    # Force re-fetch by setting expires_at to past
    t._expires_at = 0  # noqa: SLF001 — test-only
    assert await t.get() == "tok-2"
    assert fetcher.calls == 2


@pytest.mark.asyncio
async def test_refresher_concurrent_calls_share_one_fetch():
    """Two concurrent get() calls during an in-flight refresh reuse the result."""
    import asyncio

    started = asyncio.Event()
    release = asyncio.Event()

    class SlowFetcher:
        def __init__(self):
            self.calls = 0

        async def __call__(self):
            self.calls += 1
            started.set()
            await release.wait()
            return f"tok-{self.calls}"

    fetcher = SlowFetcher()
    t = CachedToken(fetcher=fetcher, ttl_seconds=7200)
    # Pre-expire so both get() will need to fetch
    t._expires_at = 0  # noqa: SLF001
    a, b = await asyncio.gather(t.get(), t.get())
    assert a == b
    assert fetcher.calls == 1
    release.set()
```

**Step 2-5:** TDD 走完(略, 模式同 v1.5.5)

**实现要点:**
- `CachedToken` 持有 `(value, expires_at)`,get() 走 lock
- TTL 到期或返回 401 → 调 fetcher 重取
- 用 `asyncio.Event` 实现"in-flight refresh 共享"模式,避免雪崩

**Step 6: 提交**

```bash
git commit -m "feat(connectors): add token-refresh helper with single-flight guarantee"
```

```python
# Feishu: 改前
self._token: str | None = None
async def _get_token(self) -> str:
    if self._token:
        return self._token
    # ... fetch

# Feishu: 改后
self._token_cache: CachedToken | None = None
async def _get_token(self) -> str:
    if self._token_cache is None:
        self._token_cache = CachedToken(
            fetcher=self._fetch_token,
            ttl_seconds=5400,  # Feishu 给 2h,提前 15min 刷新
        )
    return await self._token_cache.get()
```

并在 send() 中捕获 401 + 清空 cache + 重试 1 次(运行时已经有重试机制,只需让 token fetcher 自己刷新):

**Step 2-5: 实现**

DLQ schema:
```sql
CREATE TABLE dlq_entries (
    id TEXT PRIMARY KEY,
    connector TEXT NOT NULL,
    last_error TEXT NOT NULL,
    attempts INTEGER NOT NULL,
    message_json TEXT NOT NULL,  -- ConnectorMessage serialized
    created_at TEXT NOT NULL
);
CREATE INDEX idx_dlq_created ON dlq_entries(created_at DESC);
```

Subscription schema:
```sql
CREATE TABLE subscriptions (
    id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL,
    target_kind TEXT NOT NULL,
    target_id TEXT NOT NULL,
    notify_via_json TEXT NOT NULL,  -- list of channel values
    created_at TEXT NOT NULL
);
CREATE INDEX idx_sub_target ON subscriptions(target_kind, target_id);
```

两个 store 类都通过 `asyncpg`-free SQLAlchemy Core 写法:

```python
class SqliteDLQStore:
    def __init__(self, engine):
        self._engine = engine

    async def create_tables(self):
        async with self._engine.begin() as conn:
            await conn.execute(text("""
                CREATE TABLE IF NOT EXISTS dlq_entries (...)
            """))

    async def push(self, entry: DLQEntry) -> str:
        async with self._engine.begin() as conn:
            await conn.execute(
                text("INSERT INTO dlq_entries VALUES (:id, ...)"),
                {...},
            )
        return entry_id
```
