```python
"""Quota enforcement service."""
from __future__ import annotations

from datetime import datetime
from typing import Protocol

from .models import Quota, UsageRecord


class QuotaStore(Protocol):
    async def get_quota(self, workspace_id: str) -> Quota | None: ...
    async def set_quota(self, quota: Quota) -> None: ...
    async def get_usage(self, workspace_id: str, month: str) -> UsageRecord: ...
    async def add_usage(self, record: UsageRecord) -> None: ...


class InMemoryQuotaStore:
    def __init__(self):
        self._quotas: dict[str, Quota] = {}
        self._usage: dict[tuple[str, str], UsageRecord] = {}

    async def get_quota(self, workspace_id: str) -> Quota | None:
        return self._quotas.get(workspace_id)

    async def set_quota(self, quota: Quota) -> None:
        self._quotas[quota.workspace_id] = quota

    async def get_usage(self, workspace_id: str, month: str) -> UsageRecord:
        key = (workspace_id, month)
        if key not in self._usage:
            self._usage[key] = UsageRecord(workspace_id=workspace_id, month=month)
        return self._usage[key]

    async def add_usage(self, record: UsageRecord) -> None:
        key = (record.workspace_id, record.month)
        self._usage[key] = record


def _current_month() -> str:
    return datetime.utcnow().strftime("%Y-%m")


class QuotaService:
    def __init__(self, store: QuotaStore):
        self._store = store

    async def check_quota(self, workspace_id: str, requested_tokens: int) -> bool:
        quota = await self._store.get_quota(workspace_id)
        if quota is None:
            return False
        usage = await self._store.get_usage(workspace_id, _current_month())
        return (usage.tokens_used + requested_tokens) <= quota.monthly_token_limit

    async def record_usage(self, workspace_id: str, tokens: int) -> None:
        usage = await self._store.get_usage(workspace_id, _current_month())
        usage.add_tokens(tokens)
        await self._store.add_usage(usage)

    async def get_usage_summary(self, workspace_id: str) -> dict:
        quota = await self._store.get_quota(workspace_id)
        usage = await self._store.get_usage(workspace_id, _current_month())
        return {
            "workspace_id": workspace_id,
            "month": usage.month,
            "tokens_used": usage.tokens_used,
            "monthly_token_limit": quota.monthly_token_limit if quota else 0,
            "utilization": (usage.tokens_used / quota.monthly_token_limit) if quota and quota.monthly_token_limit > 0 else 0,
        }
```

```python
"""Model cost calculator. Prices in USD per 1K tokens.

Update MODEL_COSTS to match current provider pricing.
"""
from __future__ import annotations

# (input_cost_per_1k, output_cost_per_1k) in USD
MODEL_COSTS: dict[str, tuple[float, float]] = {
    "gpt-4o": (0.0025, 0.01),
    "gpt-4o-mini": (0.00015, 0.0006),
    "claude-opus-4-7": (0.015, 0.075),
    "claude-sonnet-4-6": (0.003, 0.015),
    "claude-haiku-4-5": (0.0008, 0.004),
    "deepseek-v3": (0.00027, 0.0011),
    "glm-4.6": (0.0006, 0.0022),
    "_default": (0.001, 0.003),  # fallback
}


class ModelCostCalculator:
    def calculate(self, model_id: str, input_tokens: int, output_tokens: int) -> float:
        if input_tokens < 0 or output_tokens < 0:
            raise ValueError("token counts must be non-negative")
        in_cost, out_cost = MODEL_COSTS.get(model_id, MODEL_COSTS["_default"])
        return (input_tokens / 1000) * in_cost + (output_tokens / 1000) * out_cost
```

```python
"""Model router: picks model based on tier, prompt prefix, or cost strategy."""
from __future__ import annotations

from dataclasses import dataclass
from enum import Enum

from .cost_calculator import MODEL_COSTS
from .models import ModelTier


class RoutingStrategy(str, Enum):
    EXPLICIT = "explicit"        # User requested a tier
    COST_OPTIMAL = "cost_optimal"  # Pick cheapest available
    QUALITY_OPTIMAL = "quality"    # Pick most expensive (best quality)
    BALANCED = "balanced"


@dataclass(frozen=True)
class RoutingRule:
    pattern: str   # "tier:fast" or "tier:premium" or any string matched against prompt prefix
    target_model: str


class ModelRouter:
    def __init__(self, rules: list[RoutingRule], default_tier: ModelTier = ModelTier.BALANCED):
        self._rules = {self._tier_to_pattern(default_tier): next(
            (r for r in rules if r.pattern == self._tier_to_pattern(default_tier)),
            RoutingRule(pattern="", target_model="claude-sonnet-4-6"),
        )}
        for r in rules:
            self._rules[r.pattern] = r

    def _tier_to_pattern(self, tier: ModelTier) -> str:
        return f"tier:{tier.value}"

    def route(
        self,
        requested_tier: ModelTier | None = None,
        prompt_prefix: str = "",
        strategy: RoutingStrategy = RoutingStrategy.EXPLICIT,
    ) -> str:
        # Explicit tier takes precedence
        if requested_tier is not None:
            rule = self._rules.get(self._tier_to_pattern(requested_tier))
            if rule:
                return rule.target_model

        # Pattern match in prompt prefix
        for pattern, rule in self._rules.items():
            if prompt_prefix.startswith(pattern):
                return rule.target_model

        # Strategy-based
        if strategy == RoutingStrategy.COST_OPTIMAL:
            return min(self._rules.values(), key=lambda r: self._avg_cost(r.target_model)).target_model
        if strategy == RoutingStrategy.QUALITY_OPTIMAL:
            return max(self._rules.values(), key=lambda r: self._avg_cost(r.target_model)).target_model

        # Default tier
        default_rule = self._rules[self._tier_to_pattern(ModelTier.BALANCED)]
        return default_rule.target_model

    def _avg_cost(self, model_id: str) -> float:
        in_c, out_c = MODEL_COSTS.get(model_id, MODEL_COSTS["_default"])
        return (in_c + out_c) / 2
```

```python
"""Track per-workspace, per-user token usage for cost dashboards."""
from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from typing import Protocol


@dataclass
class UsageEntry:
    workspace_id: str
    user_id: str
    model_id: str
    input_tokens: int
    output_tokens: int
    timestamp: str = field(default_factory=lambda: datetime.utcnow().isoformat())


class UsageStore(Protocol):
    async def record(self, entry: UsageEntry) -> None: ...
    async def query(self, workspace_id: str) -> list[UsageEntry]: ...


class InMemoryUsageStore:
    def __init__(self):
        self._entries: list[UsageEntry] = []

    async def record(self, entry: UsageEntry) -> None:
        self._entries.append(entry)

    async def query(self, workspace_id: str) -> list[UsageEntry]:
        return [e for e in self._entries if e.workspace_id == workspace_id]


class UsageTracker:
    def __init__(self, store: UsageStore):
        self._store = store

    async def record(self, workspace_id: str, user_id: str, model_id: str, input_tokens: int, output_tokens: int) -> None:
        await self._store.record(UsageEntry(
            workspace_id=workspace_id, user_id=user_id, model_id=model_id,
            input_tokens=input_tokens, output_tokens=output_tokens,
        ))

    async def summary_by_workspace(self, workspace_id: str) -> dict:
        entries = await self._store.query(workspace_id)
        return {
```

```python
"""ABAC evaluator with attribute-based policies."""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any

from .policies import AttributePolicy


@dataclass
class Subject:
    id: str
    attributes: dict[str, Any] = field(default_factory=dict)


@dataclass
class Resource:
    type: str
    id: str
    attributes: dict[str, Any] = field(default_factory=dict)


@dataclass
class Action:
    name: str


class ABACEvaluator:
    def __init__(self, policies: list[AttributePolicy]):
        self._policies = policies

    def evaluate(self, subject: Subject, resource: Resource, action_name: str) -> bool:
        action = Action(name=action_name)
        for policy in self._policies:
            if self._matches(policy, resource, action):
                try:
                    if policy.condition(subject, resource, action):
                        return True
                except Exception:
                    continue  # policy eval failed; skip
        return False

    def _matches(self, policy: AttributePolicy, resource: Resource, action: Action) -> bool:
        action_match = policy.action == "*" or policy.action == action.name
        resource_match = policy.resource_type == "*" or policy.resource_type == resource.type
        return action_match and resource_match
```

```python
"""OpenTelemetry tracing setup."""
from __future__ import annotations

import functools
from typing import Callable

from opentelemetry import trace
from opentelemetry.trace import Tracer, TracerProvider


_provider: TracerProvider | None = None


def init_tracing(provider: TracerProvider | None = None) -> None:
    """Initialize global tracer provider. Idempotent."""
    global _provider
    if provider is not None:
        _provider = provider
        trace.set_tracer_provider(provider)
    elif _provider is None:
        _provider = TracerProvider()
        trace.set_tracer_provider(_provider)


def get_tracer(name: str = "micx") -> Tracer:
    return trace.get_tracer(name)


def traced(span_name: str) -> Callable:
    """Decorator: wrap function in an OTel span."""
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            tracer = get_tracer()
            with tracer.start_as_current_span(span_name) as span:
                try:
                    result = func(*args, **kwargs)
                    return result
                except Exception as e:
                    span.record_exception(e)
                    span.set_status(trace.Status(trace.StatusCode.ERROR, str(e)))
                    raise
        return wrapper
    return decorator
```

```python
"""Langfuse trace exporter. Sends OTel spans to Langfuse v2 ingestion API."""
from __future__ import annotations

import httpx


class LangfuseExporter:
    def __init__(self, public_key: str, secret_key: str, host: str, timeout: float = 10.0):
        if not public_key:
            raise ValueError("public_key required")
        if not secret_key:
            raise ValueError("secret_key required")
        self.host = host.rstrip("/")
        self._auth = (public_key, secret_key)
        self._timeout = timeout

    async def export_batch(self, spans: list[dict]) -> bool:
        """Send a batch of OTel-formatted spans to Langfuse."""
        if not spans:
            return True
        payload = {"batch": spans}
        async with httpx.AsyncClient(timeout=self._timeout) as client:
            try:
                resp = await client.post(
                    f"{self.host}/api/public/ingestion",
                    json=payload,
                    auth=self._auth,
                )
                return resp.status_code in (200, 201, 207)
            except httpx.HTTPError:
                return False
```

```python
"""In-memory workflow store. Production: DB-backed."""
from __future__ import annotations

from typing import Protocol

from .models import Workflow, WorkflowStatus


class WorkflowStore(Protocol):
    async def save(self, wf: Workflow) -> None: ...
    async def get(self, wf_id: str) -> Workflow | None: ...
    async def list_by_status(self, status: WorkflowStatus) -> list[Workflow]: ...


class InMemoryWorkflowStore:
    def __init__(self):
        self._items: dict[str, Workflow] = {}

    async def save(self, wf: Workflow) -> None:
        self._items[wf.id] = wf

    async def get(self, wf_id: str) -> Workflow | None:
        return self._items.get(wf_id)

    async def list_by_status(self, status: WorkflowStatus) -> list[Workflow]:
        return [w for w in self._items.values() if w.status == status]
```

```python
"""Branch node: evaluates a simple condition and routes to true/false edge.

For v1.6.0, supports a minimal expression syntax: 'var op value' (e.g. 'x > 5').
Real DSL deferred to v1.7+.
"""
from __future__ import annotations

import re
from typing import Any

from ..models import WorkflowNode

EXPR_RE = re.compile(r"^(\w+)\s*(>|>=|<|<=|==|!=)\s*(.+)$")


class BranchNode:
    def __init__(self, node: WorkflowNode):
        self.node = node

    async def execute(self, inputs: dict[str, Any]) -> dict[str, Any]:
        condition = self.node.config.get("condition", "")
        m = EXPR_RE.match(condition)
        if not m:
            return {"branch": "false", "reason": "invalid expression"}

        var_name, op, raw_value = m.groups()
        var_value = inputs.get(var_name)
        try:
            value = float(raw_value)
        except ValueError:
            value = raw_value.strip().strip('"').strip("'")

        try:
            var_num = float(var_value)
            value_num = float(value)
            result = self._compare(var_num, op, value_num)
        except (ValueError, TypeError):
            result = self._compare(str(var_value), op, str(value))

        return {"branch": "true" if result else "false", "expression": condition}

    def _compare(self, a, op, b) -> bool:
        if op == ">": return a > b
        if op == ">=": return a >= b
        if op == "<": return a < b
        if op == "<=": return a <= b
        if op == "==": return a == b
        if op == "!=": return a != b
        return False
```

```python
"""Sequential workflow executor (v1.6.0 MVP).

Executes nodes in topological order. Branch/loop supported via
specialized node executors.
"""
from __future__ import annotations

from typing import Any

from .models import Workflow, WorkflowNode
from .nodes.agent import AgentNode
from .nodes.tool import ToolNode
from .nodes.branch import BranchNode
from .nodes.loop import LoopNode
from .store import WorkflowStore


class WorkflowExecutor:
    def __init__(self, store: WorkflowStore):
        self._store = store

    async def execute(self, workflow_id: str, initial_inputs: dict[str, Any]) -> dict[str, Any]:
        wf = await self._store.get(workflow_id)
        if wf is None:
            raise ValueError(f"workflow {workflow_id!r} not found")

        # Topological sort: in MVP, assume nodes are in execution order
        outputs: dict[str, Any] = {}
        current_inputs = dict(initial_inputs)
        for node in wf.nodes:
            out = await self._execute_node(node, current_inputs)
            outputs[node.id] = out
            current_inputs.update(out)
        return {"outputs": outputs, "workflow_id": wf.id}

    async def _execute_node(self, node: WorkflowNode, inputs: dict[str, Any]) -> dict[str, Any]:
        if node.kind.value == "agent":
            return await AgentNode(node).execute(inputs)
        if node.kind.value == "tool":
            return await ToolNode(node).execute(inputs)
        if node.kind.value == "branch":
            return await BranchNode(node).execute(inputs)
        if node.kind.value == "loop":
            async def body(i, _):
                return {"iteration": i}
            return await LoopNode(node).execute(inputs, body=body)
        if node.kind.value == "prompt":
            return {"output": node.config.get("text", "")}
        return {"error": f"unknown node kind: {node.kind}"}
```

```python
"""Workflow version management with rollback."""
from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from typing import Any, Protocol


@dataclass
class WorkflowVersion:
    workflow_id: str
    version: int
    snapshot: dict[str, Any]
    created_at: str = field(default_factory=lambda: datetime.utcnow().isoformat())


class VersionStore(Protocol):
    async def save(self, v: WorkflowVersion) -> None: ...
    async def list_versions(self, workflow_id: str) -> list[WorkflowVersion]: ...
    async def get_version(self, workflow_id: str, version: int) -> WorkflowVersion | None: ...


class InMemoryVersionStore:
    def __init__(self):
        self._items: dict[tuple[str, int], WorkflowVersion] = {}

    async def save(self, v: WorkflowVersion) -> None:
        self._items[(v.workflow_id, v.version)] = v

    async def list_versions(self, workflow_id: str) -> list[WorkflowVersion]:
        return sorted(
            [v for (wid, _), v in self._items.items() if wid == workflow_id],
            key=lambda v: v.version,
        )

    async def get_version(self, workflow_id: str, version: int) -> WorkflowVersion | None:
        return self._items.get((workflow_id, version))


class VersionManager:
    def __init__(self, store: VersionStore):
        self._store = store

    async def commit(self, workflow_id: str, version: int, snapshot: dict[str, Any]) -> WorkflowVersion:
        v = WorkflowVersion(workflow_id=workflow_id, version=version, snapshot=snapshot)
        await self._store.save(v)
        return v

    async def list_versions(self, workflow_id: str) -> list[WorkflowVersion]:
        return await self._store.list_versions(workflow_id)

    async def rollback(self, workflow_id: str, target_version: int) -> dict[str, Any]:
        v = await self._store.get_version(workflow_id, target_version)
        if v is None:
            raise ValueError(f"version {target_version} of {workflow_id!r} not found")
        return v.snapshot
```

```python
"""Cross-department thread publishing."""
from __future__ import annotations

import uuid
from dataclasses import dataclass, field


@dataclass
class PublishRecord:
    id: str = field(default_factory=lambda: str(uuid.uuid4()))
    source_thread_id: str = ""
    target_workspace_id: str = ""
    actor_id: str = ""
    original_thread_id: str = ""  # chain-preserved origin
    published_thread_id: str = field(default_factory=lambda: str(uuid.uuid4()))


class InMemoryPublishStore:
    def __init__(self):
        self._items: list[PublishRecord] = []

    async def add(self, r: PublishRecord) -> None:
        self._items.append(r)

    async def get_by_source(self, source_thread_id: str) -> list[PublishRecord]:
        return [r for r in self._items if r.source_thread_id == source_thread_id]


class PublishService:
    def __init__(self, store: InMemoryPublishStore):
        self._store = store

    async def publish(self, source_thread_id: str, target_workspace_id: str, actor_id: str) -> PublishRecord:
        # Trace chain back to original
        history = await self._store.get_by_source(source_thread_id)
        original = history[0].source_thread_id if history else source_thread_id
        r = PublishRecord(
            source_thread_id=source_thread_id,
            target_workspace_id=target_workspace_id,
            actor_id=actor_id,
            original_thread_id=original,
        )
        await self._store.add(r)
        return r
```

```python
"""Jira connector (Atlassian Cloud / Server)."""
from __future__ import annotations

import base64

import httpx

from ..base import BaseConnector, ConnectorMessage, ConnectorResponse


class JiraConnector(BaseConnector):
    name = "jira"
    display_name = "Jira"

    def __init__(self, base_url: str, email: str, api_token: str, timeout: float = 10.0):
        self._base = base_url.rstrip("/")
        auth_str = f"{email}:{api_token}"
        self._auth_header = "Basic " + base64.b64encode(auth_str.encode("utf-8")).decode("ascii")
        self._timeout = timeout

    async def send(self, message: ConnectorMessage) -> ConnectorResponse:
        try:
            target = message.target
            project = target.get("project_key", "")
            issue_type = target.get("issue_type", "Task")
            async with httpx.AsyncClient(timeout=self._timeout) as client:
                resp = await client.post(
                    f"{self._base}/rest/api/3/issue",
                    headers={
                        "Authorization": self._auth_header,
                        "Content-Type": "application/json",
                    },
                    json={
                        "fields": {
                            "project": {"key": project},
                            "summary": message.text,
                            "issuetype": {"name": issue_type},
                        }
                    },
                )
            data = resp.json()
            if resp.status_code in (200, 201) and "key" in data:
                return ConnectorResponse(success=True, external_id=data["key"], raw=data)
            return ConnectorResponse(success=False, error=str(data))
        except Exception as e:
            return ConnectorResponse(success=False, error=str(e))

    async def receive_webhook(self, payload: dict) -> list[ConnectorMessage]:
        event = payload.get("webhookEvent", "")
        issue = payload.get("issue", {})
        key = issue.get("key", "")
        summary = issue.get("fields", {}).get("summary", "")
        if not event.startswith("jira:") or not key:
            return []
        return [
            ConnectorMessage(
                text=f"[{event}] {key}: {summary}",
                target={"issue_key": key},
                metadata={"event": event},
            )
        ]
```

```python
LINEAR_API = "https://api.linear.app/graphql"


class LinearConnector(BaseConnector):
    name = "linear"
    display_name = "Linear"

    def __init__(self, api_key: str, team_id: str, timeout: float = 10.0):
        self._api_key = api_key
        self._team_id = team_id
        self._timeout = timeout

    async def send(self, message: ConnectorMessage) -> ConnectorResponse:
        try:
            title = message.target.get("title", message.text)
            query = """
            mutation IssueCreate($title: String!, $teamId: String!) {
              issueCreate(input: { title: $title, teamId: $teamId }) {
                success
                issue { id identifier }
              }
            }
            """
            async with httpx.AsyncClient(timeout=self._timeout) as client:
                resp = await client.post(
                    LINEAR_API,
                    headers={"Authorization": self._api_key, "Content-Type": "application/json"},
                    json={"query": query, "variables": {"title": title, "teamId": self._team_id}},
                )
            data = resp.json()
            if data.get("data", {}).get("issueCreate", {}).get("success"):
                issue = data["data"]["issueCreate"]["issue"]
                return ConnectorResponse(success=True, external_id=issue["identifier"], raw=data)
            return ConnectorResponse(success=False, error=str(data))
        except Exception as e:
            return ConnectorResponse(success=False, error=str(e))

    async def receive_webhook(self, payload: dict) -> list[ConnectorMessage]:
        action = payload.get("action", "")
        data = payload.get("data", {})
        identifier = data.get("identifier", "")
        title = data.get("title", "")
        if not action or not identifier:
            return []
        return [
            ConnectorMessage(
                text=f"[Linear {action}] {identifier}: {title}",
                target={"issue_id": data.get("id", "")},
                metadata={"action": action},
            )
        ]
```

```tsx
    id: n.id, position: { x: 100, y: 100 + i * 100 },
    data: { label: `${n.kind}: ${n.config?.prompt || n.config?.tool || ''}` },
  }));
  const initialEdges = wf.edges.map((e) => ({ id: e.id, source: e.source, target: e.target }));

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold mb-4">{wf.name} (v{wf.version})</h1>
      <Canvas initialNodes={initialNodes} initialEdges={initialEdges} />
    </div>
  );
}
```
