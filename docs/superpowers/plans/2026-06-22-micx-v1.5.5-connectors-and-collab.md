```python
"""Base connector protocol and data types."""
from __future__ import annotations

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Any


@dataclass
class ConnectorMessage:
    """Outgoing or incoming message."""
    text: str = ""
    target: dict = field(default_factory=dict)  # {chat_id, user_id, thread_id, ...}
    attachments: list[dict] = field(default_factory=list)
    metadata: dict = field(default_factory=dict)


@dataclass
class ConnectorResponse:
    """Result of send() or webhook processing."""
    success: bool
    external_id: str | None = None
    error: str | None = None
    raw: Any = None


class BaseConnector(ABC):
    """Abstract connector. Subclasses implement send + receive_webhook."""

    name: str = ""
    display_name: str = ""

    @abstractmethod
    async def send(self, message: ConnectorMessage) -> ConnectorResponse: ...

    @abstractmethod
    async def receive_webhook(self, payload: dict) -> list[ConnectorMessage]: ...

    async def health_check(self) -> bool:
        """Default: returns True. Override for connectivity checks."""
        return True
```

```python
"""In-process registry of connector instances."""
from __future__ import annotations

from threading import RLock

from .base import BaseConnector


class ConnectorRegistry:
    def __init__(self):
        self._connectors: dict[str, BaseConnector] = {}
        self._lock = RLock()

    def register(self, connector: BaseConnector) -> None:
        with self._lock:
            self._connectors[connector.name] = connector

    def unregister(self, name: str) -> None:
        with self._lock:
            self._connectors.pop(name, None)

    def get(self, name: str) -> BaseConnector:
        with self._lock:
            if name not in self._connectors:
                raise KeyError(f"connector {name!r} not registered")
            return self._connectors[name]

    def list_names(self) -> list[str]:
        with self._lock:
            return list(self._connectors.keys())


_registry: ConnectorRegistry | None = None


def get_registry() -> ConnectorRegistry:
    global _registry
    if _registry is None:
        _registry = ConnectorRegistry()
    return _registry
```

```python
"""YAML config schema for connectors."""
from __future__ import annotations

SUPPORTED_TYPES = frozenset({"feishu", "dingtalk", "wecom", "email", "slack"})


class ConfigError(ValueError):
    """Connector config validation failed."""


def validate_config(cfg: dict) -> None:
    """Validate a parsed YAML config dict. Raises ConfigError on issue."""
    if not isinstance(cfg, dict):
        raise ConfigError("config must be a mapping")

    for required in ("name", "type", "enabled", "credentials"):
        if required not in cfg:
            raise ConfigError(f"missing required field: {required!r}")

    if cfg["type"] not in SUPPORTED_TYPES:
        raise ConfigError(f"unknown type: {cfg['type']!r}; supported: {sorted(SUPPORTED_TYPES)}")

    if not isinstance(cfg["credentials"], dict):
        raise ConfigError("credentials must be a mapping")

    if not isinstance(cfg["enabled"], bool):
        raise ConfigError("enabled must be a boolean")

    # routes is optional
    if "routes" in cfg and not isinstance(cfg["routes"], list):
        raise ConfigError("routes must be a list")
```

```python
"""Connector runtime: executes send, retries on failure, routes to DLQ."""
from __future__ import annotations

import asyncio
from dataclasses import dataclass, field

from .base import BaseConnector, ConnectorMessage, ConnectorResponse


@dataclass
class DLQEntry:
    connector: str
    message: ConnectorMessage
    last_error: str
    attempts: int
    timestamp: str = ""


@dataclass
class RuntimeResponse:
    success: bool
    external_id: str | None = None
    error: str | None = None
    attempts: int = 0
    dlq: DLQEntry | None = None


class ConnectorRuntime:
    def __init__(
        self,
        max_retries: int = 2,
        retry_delay: float = 1.0,
        dlq_enabled: bool = True,
    ):
        self._connectors: dict[str, BaseConnector] = {}
        self._max_retries = max_retries
        self._retry_delay = retry_delay
        self._dlq_enabled = dlq_enabled
        self._dlq: list[DLQEntry] = []

    def register(self, connector: BaseConnector) -> None:
        self._connectors[connector.name] = connector

    async def send(self, connector_name: str, message: ConnectorMessage) -> RuntimeResponse:
        if connector_name not in self._connectors:
            return RuntimeResponse(success=False, error=f"unknown connector: {connector_name}")
        connector = self._connectors[connector_name]
        last_error = ""
        for attempt in range(1, self._max_retries + 2):  # 1 initial + N retries
            try:
                resp = await connector.send(message)
                if resp.success:
                    return RuntimeResponse(
                        success=True, external_id=resp.external_id, attempts=attempt,
                    )
                last_error = resp.error or "unknown"
            except Exception as e:
                last_error = str(e)
            if attempt <= self._max_retries:
                await asyncio.sleep(self._retry_delay)

        # All attempts failed
        dlq_entry: DLQEntry | None = None
        if self._dlq_enabled:
            dlq_entry = DLQEntry(
                connector=connector_name, message=message,
                last_error=last_error, attempts=self._max_retries + 1,
            )
            self._dlq.append(dlq_entry)
        return RuntimeResponse(
            success=False, error=last_error,
            attempts=self._max_retries + 1, dlq=dlq_entry,
        )

    def get_dlq(self) -> list[DLQEntry]:
        return list(self._dlq)

    def clear_dlq(self) -> int:
        n = len(self._dlq)
        self._dlq.clear()
        return n
```

```python
"""Dead-letter queue storage. In-memory for now; production: DB-backed."""
from __future__ import annotations

import time
import uuid


class InMemoryDLQStore:
    def __init__(self):
        self._items: dict[str, dict] = {}

    def push(self, entry: dict) -> str:
        item_id = str(uuid.uuid4())
        entry = {
            "id": item_id,
            "timestamp": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
            **entry,
        }
        self._items[item_id] = entry
        return item_id

    def list_all(self) -> list[dict]:
        return sorted(self._items.values(), key=lambda x: x["timestamp"], reverse=True)

    def get(self, item_id: str) -> dict | None:
        return self._items.get(item_id)

    def delete(self, item_id: str) -> bool:
        return self._items.pop(item_id, None) is not None

    def clear_all(self) -> int:
        n = len(self._items)
        self._items.clear()
        return n
```

```python
"""Webhook bridge: receives inbound webhooks, dispatches to connectors."""
from __future__ import annotations

from .base import BaseConnector, ConnectorMessage


class WebhookBridge:
    def __init__(self):
        self._routes: dict[str, tuple[BaseConnector, str]] = {}

    def register(self, connector: BaseConnector, secret: str) -> None:
        self._routes[connector.name] = (connector, secret)

    async def handle_inbound(
        self, connector_name: str, secret: str, payload: dict,
    ) -> list[ConnectorMessage]:
        if connector_name not in self._routes:
            raise KeyError(f"connector {connector_name!r} not registered")
        conn, expected_secret = self._routes[connector_name]
        if secret != expected_secret:
            raise PermissionError("invalid webhook secret")
        return await conn.receive_webhook(payload)
```

```python
"""Feishu (Lark) connector.

Implements BaseConnector for Feishu OpenAPI:
- Outbound: send message to chat / user
- Inbound: receive event webhook
"""
from __future__ import annotations

import httpx

from ..base import BaseConnector, ConnectorMessage, ConnectorResponse

FEISHU_BASE = "https://open.feishu.cn/open-apis"


class FeishuConnector(BaseConnector):
    name = "feishu"
    display_name = "Feishu (Lark)"

    def __init__(self, app_id: str, app_secret: str, timeout: float = 10.0):
        self._app_id = app_id
        self._app_secret = app_secret
        self._timeout = timeout
        self._token: str | None = None

    async def _get_token(self) -> str:
        async with httpx.AsyncClient(timeout=self._timeout) as client:
            resp = await client.post(
                f"{FEISHU_BASE}/auth/v3/tenant_access_token/internal",
                json={"app_id": self._app_id, "app_secret": self._app_secret},
            )
            data = resp.json()
            if data.get("code") != 0:
                raise RuntimeError(f"feishu token error: {data}")
            self._token = data["tenant_access_token"]
            return self._token

    async def send(self, message: ConnectorMessage) -> ConnectorResponse:
        try:
            token = await self._get_token()
            receive_id = message.target.get("chat_id") or message.target.get("user_id")
            receive_id_type = "chat_id" if message.target.get("chat_id") else "open_id"
            async with httpx.AsyncClient(timeout=self._timeout) as client:
                resp = await client.post(
                    f"{FEISHU_BASE}/im/v1/messages",
                    headers={"Authorization": f"Bearer {token}"},
                    json={
                        "receive_id": receive_id,
                        "msg_type": "text",
                        "content": '{"text": "' + message.text.replace('"', '\\"') + '"}',
                    },
                    params={"receive_id_type": receive_id_type},
                )
            data = resp.json()
            if data.get("code") == 0:
                return ConnectorResponse(
                    success=True, external_id=data["data"]["message_id"], raw=data,
                )
            return ConnectorResponse(success=False, error=data.get("msg", "unknown"))
        except Exception as e:
            return ConnectorResponse(success=False, error=str(e))

    async def receive_webhook(self, payload: dict) -> list[ConnectorMessage]:
        if payload.get("type") == "url_verification":
            return []
        event = payload.get("event", {})
        text = event.get("text", {}).get("plain_text", "") if isinstance(event.get("text"), dict) else ""
        if not text:
            return []
        return [
            ConnectorMessage(
                text=text,
                target={"chat_id": event.get("chat_id", "")},
                metadata={"sender": event.get("sender", {}).get("sender_id", {}).get("open_id")},
            )
        ]
```

```python
        timeout: float = 15.0,
    ):
        self._smtp_host = smtp_host
        self._smtp_port = smtp_port
        self._smtp_user = smtp_user
        self._smtp_password = smtp_password
        self._imap_host = imap_host
        self._imap_port = imap_port
        self._from = from_address
        self._use_tls = use_tls
        self._timeout = timeout

    async def send(self, message: ConnectorMessage) -> ConnectorResponse:
        try:
            to = message.target.get("to", "")
            subject = message.target.get("subject", "(no subject)")
            email = EmailMessage()
            email["From"] = self._from
            email["To"] = to
            email["Subject"] = subject
            email.set_content(message.text)

            await aiosmtplib.send(
                email,
                hostname=self._smtp_host,
                port=self._smtp_port,
                username=self._smtp_user,
                password=self._smtp_password,
                use_tls=self._use_tls,
                timeout=self._timeout,
            )
            return ConnectorResponse(success=True, external_id=email["Message-ID"])
        except Exception as e:
            return ConnectorResponse(success=False, error=str(e))

    async def receive_webhook(self, payload: dict) -> list[ConnectorMessage]:
        """Email doesn't use webhooks in the traditional sense. Inbound emails
        are typically polled via IMAP IDLE; that's a separate concern handled
        by a poller worker. For now, return empty list."""
        return []
```

```python
"""SAML provider wrapper around python3-saml."""
from __future__ import annotations

from dataclasses import dataclass
from onelogin.saml2.auth import OneLogin_Saml2_Auth
from onelogin.saml2.settings import OneLogin_Saml2_Settings
from onelogin.saml2.utils import OneLogin_Saml2_Utils

from .idp import SAMLUserInfo


@dataclass(frozen=True)
class SAMLConfig:
    idp_entity_id: str
    idp_sso_url: str
    idp_x509_cert: str
    sp_entity_id: str
    sp_acs_url: str

    def __post_init__(self):
        if not self.idp_entity_id:
            raise ValueError("idp_entity_id required")
        if not self.idp_sso_url:
            raise ValueError("idp_sso_url required")
        if not self.idp_x509_cert:
            raise ValueError("idp_x509_cert required")
        if not self.sp_entity_id:
            raise ValueError("sp_entity_id required")
        if not self.sp_acs_url:
            raise ValueError("sp_acs_url required")


class SAMLProvider:
    def __init__(self, config: SAMLConfig):
        self.config = config

    def get_sp_metadata(self) -> str:
        settings_dict = self._build_settings_dict()
        settings = OneLogin_Saml2_Settings(settings_dict, sp_validation_only=True)
        metadata = settings.get_sp_metadata()
        errors = settings.validate_metadata(metadata)
        if errors:
            raise RuntimeError(f"SP metadata invalid: {errors}")
        return metadata

    def _build_settings_dict(self) -> dict:
        return {
            "sp": {
                "entityId": self.config.sp_entity_id,
                "assertionConsumerService": {
                    "url": self.config.sp_acs_url,
                    "binding": "urn:oasis:names:tc:SAML:2.0:bindings:HTTP-POST",
                },
            },
            "idp": {
                "entityId": self.config.idp_entity_id,
                "singleSignOnService": {
                    "url": self.config.idp_sso_url,
                    "binding": "urn:oasis:names:tc:SAML:2.0:bindings:HTTP-Redirect",
                },
                "x509cert": self.config.idp_x509_cert,
            },
        }

    def process_acs(self, request_data: dict) -> SAMLUserInfo:
        settings_dict = self._build_settings_dict()
        auth = OneLogin_Saml2_Auth(request_data, settings_dict)
        auth.process_response()
        errors = auth.get_errors()
        if errors:
            raise RuntimeError(f"SAML response error: {errors}")
        if not auth.is_authenticated():
            raise RuntimeError("SAML response not authenticated")
        attributes = auth.get_attributes()
        return SAMLUserInfo(
            name_id=auth.get_nameid(),
            email=(attributes.get("email") or [""])[0],
            name=(attributes.get("displayName") or attributes.get("name") or [""])[0],
            groups=attributes.get("groups") or attributes.get("memberOf") or [],
        )
```

```python
"""SAML Assertion Consumer Service endpoint."""
from __future__ import annotations

from fastapi import APIRouter, Request
from fastapi.responses import RedirectResponse

from .provider import SAMLProvider

router = APIRouter(prefix="/auth/saml", tags=["auth-saml"])

# In-process provider registry; production: DB-backed
_providers: dict[str, SAMLProvider] = {}


def register_saml_provider(name: str, provider: SAMLProvider) -> None:
    _providers[name] = provider


@router.post("/acs/{provider_name}")
async def acs(provider_name: str, request: Request) -> RedirectResponse:
    if provider_name not in _providers:
        return RedirectResponse(f"/sign-in?error=unknown_saml_provider", status_code=303)
    provider = _providers[provider_name]
    form = await request.form()
    request_data = {
        "https": "on" if request.url.scheme == "https" else "off",
        "http_host": request.url.hostname or "",
        "script_name": request.url.path,
        "post_data": dict(form),
    }
    user_info = provider.process_acs(request_data)
    # In production: create session, redirect
    return RedirectResponse(f"/?saml_user={user_info.email}", status_code=303)
```

```python
"""Subscription service: handles @mentions and triggers notifications."""
from __future__ import annotations

from dataclasses import dataclass

from .models import Subscription, SubscriptionTarget
from .notifier import InMemoryNotifier, Notification


class SubscriptionStore:
    async def get_for_target(self, target: SubscriptionTarget) -> list[Subscription]: ...
    async def add(self, sub: Subscription) -> None: ...


class InMemorySubscriptionStore:
    def __init__(self):
        self._items: list[Subscription] = []

    async def get_for_target(self, target: SubscriptionTarget) -> list[Subscription]:
        return [s for s in self._items if s.target == target]

    async def add(self, sub: Subscription) -> None:
        self._items.append(sub)


class SubscriptionService:
    def __init__(self, store: SubscriptionStore, notifier):
        self._store = store
        self._notifier = notifier

    async def handle_mention(
        self, thread_id: str, mentioned_user_ids: list[str], actor_id: str, text: str,
    ) -> int:
        """Send notifications for each mentioned user. Returns count sent."""
        sent = 0
        for uid in mentioned_user_ids:
            target = SubscriptionTarget(kind="user", id=uid)
            subs = await self._store.get_for_target(target)
            for sub in subs:
                await self._notifier.send(Notification(
                    recipient=uid,
                    text=f"{actor_id} mentioned you: {text[:100]}",
                    metadata={"thread_id": thread_id, "actor_id": actor_id},
                ))
                sent += 1
        return sent
```
