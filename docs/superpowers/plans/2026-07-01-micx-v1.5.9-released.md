```python
"""SQLite-backed CommentStore.

Same surface as ``InMemoryCommentStore`` (add / get / list_for_thread
/ delete). The schema lives at
``backend/app/gateway/data/migrations/004_v1.5.8_multitenancy.sql``
(comments table + idx_comments_thread index).

Switched on via env var ``MICX_COMMENTS_STORE=sqlite`` — see
``service.get_comment_store``. Default is ``"memory"`` to keep dev
cold-start behaviour identical to v1.5.8.
"""
from __future__ import annotations

import sqlite3
import threading
import uuid
from datetime import datetime, timezone

from .models import Comment, CommentSource


def _utcnow() -> str:
    return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%S.%fZ")


class SqliteCommentStore:
    """Persistent thread-scoped comment store.

    Thread-safe: a single lock guards the connection, one writer at a
    time. Read contention is non-blocking (sqlite default).
    """

    def __init__(self, db_path: str) -> None:
        self._db_path = db_path
        self._lock = threading.Lock()
        # check_same_thread=False lets FastAPI sync test code connect
        # from another thread; we still serialize via _lock.
        self._conn = sqlite3.connect(db_path, check_same_thread=False)
        self._conn.row_factory = sqlite3.Row
        self._ensure_schema()

    def _ensure_schema(self) -> None:
        with self._lock:
            self._conn.executescript(
                """
                CREATE TABLE IF NOT EXISTS comments (
                    id         TEXT PRIMARY KEY,
                    thread_id  TEXT NOT NULL,
                    author     TEXT NOT NULL,
                    body       TEXT NOT NULL,
                    parent_id  TEXT,
                    source     TEXT NOT NULL DEFAULT 'user',
                    created_at TEXT NOT NULL
                );
                CREATE INDEX IF NOT EXISTS idx_comments_thread
                    ON comments(thread_id);
                """
            )
            self._conn.commit()

    def add(
        self,
        thread_id: str,
        author_id: str,
        text: str,
        source: CommentSource = CommentSource.USER,
        parent_comment_id: str | None = None,
    ) -> Comment:
        cid = f"c-{uuid.uuid4().hex[:12]}"
        created_at = _utcnow()
        source_str = source.value if hasattr(source, "value") else str(source)
        with self._lock:
            self._conn.execute(
                "INSERT INTO comments (id, thread_id, author, body, parent_id, source, created_at) "
                "VALUES (?, ?, ?, ?, ?, ?, ?)",
                (
                    cid, thread_id, author_id, text,
                    parent_comment_id, source_str, created_at,
                ),
            )
            self._conn.commit()
        # Mentions are derived client-side; reusing the in-memory
        # extraction keeps this store honest with the in-memory one
        # so the API surface stays symmetric.
        from .service import extract_mentions
        mentioned = extract_mentions(text)
        return Comment(
            id=cid,
            thread_id=thread_id,
            author_id=author_id,
            text=text,
            source=source,
            parent_comment_id=parent_comment_id,
            mentioned_user_ids=mentioned,
            created_at=created_at,
        )

    def get(self, comment_id: str) -> Comment | None:
        with self._lock:
            row = self._conn.execute(
                "SELECT * FROM comments WHERE id = ?", (comment_id,),
            ).fetchone()
        return _row_to_comment(row) if row else None

    def list_for_thread(
        self, thread_id: str, limit: int = 100
    ) -> list[Comment]:
        with self._lock:
            rows = self._conn.execute(
                "SELECT * FROM comments WHERE thread_id = ? "
                "ORDER BY created_at DESC, id DESC LIMIT ?",
                (thread_id, limit),
            ).fetchall()
        return [_row_to_comment(r) for r in rows]

    def delete(self, comment_id: str) -> bool:
        with self._lock:
            cur = self._conn.execute(
                "DELETE FROM comments WHERE id = ?", (comment_id,),
            )
            self._conn.commit()
            return cur.rowcount > 0


def _row_to_comment(row: sqlite3.Row) -> Comment:
    return Comment(
        id=row["id"],
        thread_id=row["thread_id"],
        author_id=row["author"],
        text=row["body"],
        source=CommentSource(row["source"]),
        parent_comment_id=row["parent_id"],
        mentioned_user_ids=[],  # re-extracted at write time
        created_at=row["created_at"],
    )
```

```python
def get_comment_store():
    """Resolve the comment store based on the MICX_COMMENTS_STORE env.

    Returns ``InMemoryCommentStore`` by default (v1.5.8 behaviour, dev
    cold-start friendly). Set ``MICX_COMMENTS_STORE=sqlite`` to switch
    to ``SqliteCommentStore``; the DB path defaults to
    ``backend/.deer-flow/comments.db`` and can be overridden by
    ``MICX_COMMENTS_DB``.
    """
    import os

    backend = os.environ.get("MICX_COMMENTS_STORE", "memory").lower()
    if backend == "sqlite":
        from .sqlite_store import SqliteCommentStore

        db_path = os.environ.get(
            "MICX_COMMENTS_DB", ".deer-flow/comments.db",
        )
        return SqliteCommentStore(db_path)
    return InMemoryCommentStore()
```

```tsx
const NODE_KINDS = ["trigger", "transform", "decision", "output", "annotate"] as const;
type NodeKind = typeof NODE_KINDS[number];

function kindToNodeType(kind: NodeKind): string {
  return `node-${kind}`;
}

const NodePalette: NodeTypes = Object.fromEntries(
  NODE_KINDS.map((k) => [
    kindToNodeType(k),
    ({ data }: { data: { label?: string; kind?: NodeKind } }) => (
      <div className="rounded border bg-background p-2 text-xs">
        <div className="font-medium">{data?.label ?? data?.kind ?? ""}</div>
      </div>
    ),
  ]),
) as NodeTypes;

export function CanvasView({
  canvas,
  onChange,
}: {
  canvas: Canvas;
  onChange: (next: Canvas) => void;
}) {
  const nodes: RFNode[] = canvas.nodes.map((n) => ({
    id: n.id,
    type: kindToNodeType(n.kind),
    position: { x: n.x, y: n.y },
    data: { label: n.label, kind: n.kind },
  }));
  const edges: RFEdge[] = canvas.edges.map((e) => ({
    id: e.id,
    source: e.from,
    target: e.to,
  }));

  return (
    <ReactFlowProvider>
      <section
        data-testid="canvas-view"
        className="relative h-[60vh] w-full overflow-hidden rounded border"
      >
        <ReactFlow
          nodes={nodes}
          edges={edges}
          nodeTypes={NodePalette}
          onNodesChange={(changes) => {
            // Translate RF changes back into the existing Canvas shape.
            // For v1.5.9 we keep it minimal: position moves only.
            const next = canvas.nodes.map((n) => ({ ...n }));
            for (const ch of changes) {
              if (ch.type === "position" && ch.position && ch.id) {
                const target = next.find((x) => x.id === ch.id);
                if (target) {
                  target.x = ch.position.x;
                  target.y = ch.position.y;
                }
              }
            }
            onChange({ ...canvas, nodes: next });
          }}
          fitView
          fitViewOptions={{ padding: 0.2 }}
        >
          <Background gap={16} />
          <Controls />
        </ReactFlow>
      </section>
      {/* Existing action bar — unchanged. */}
      <div data-testid="canvas-actions" className="mt-2 flex gap-2">
        {NODE_KINDS.map((k) => (
          <button
            key={k}
            type="button"
            data-testid={`palette-${k}`}
            className="rounded border px-2 py-1 text-xs"
            onClick={() =>
              onChange({
                ...canvas,
                nodes: [
                  ...canvas.nodes,
                  {
                    id: `n-${Math.random().toString(36).slice(2, 10)}`,
                    kind: k,
                    label: k,
                    x: 80,
                    y: 80 + canvas.nodes.length * 60,
                  },
                ],
              })
            }
          >
            + {k}
          </button>
        ))}
      </div>
      {/* Node list — preserves `data-testid="node-{id}"` and `remove-{id}`. */}
      <ul className="mt-2 flex flex-col gap-1 text-xs">
        {canvas.nodes.map((n) => (
          <li
            key={n.id}
            data-testid={`node-${n.id}`}
            className="flex items-center justify-between rounded border px-2 py-1"
          >
            <span>
              {n.kind} / {n.label}
            </span>
            <button
              type="button"
              data-testid={`remove-${n.id}`}
              onClick={() =>
                onChange({
                  ...canvas,
                  nodes: canvas.nodes.filter((x) => x.id !== n.id),
                })
              }
            >
              ×
            </button>
          </li>
        ))}
      </ul>
    </ReactFlowProvider>
  );
}
```

```tsx
"use client";

import { useEffect, useState } from "react";

import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";

interface ApiWorkspace {
  id: string;
  name: string;
  plan: string;
  created_at: string;
}

export function WorkspacesAdminPage() {
  const [items, setItems] = useState<ApiWorkspace[]>([]);
  const [name, setName] = useState("");
  const [plan, setPlan] = useState("free");
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const reload = async () => {
    try {
      const r = await fetch("/api/workspaces", { credentials: "include" });
      if (r.ok) {
        const body = (await r.json()) as { workspaces: ApiWorkspace[] };
        setItems(body.workspaces ?? body as unknown as ApiWorkspace[]);
      }
    } catch (e) {
      setError(e instanceof Error ? e.message : String(e));
    }
  };

  useEffect(() => {
    void reload();
  }, []);

  const submit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!name.trim()) return;
    setLoading(true);
    setError(null);
    try {
      const r = await fetch("/api/workspaces", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        credentials: "include",
        body: JSON.stringify({ name, plan }),
      });
      if (!r.ok) throw new Error(`POST failed: ${r.status}`);
      setName("");
      await reload();
    } catch (e) {
      setError(e instanceof Error ? e.message : String(e));
    } finally {
      setLoading(false);
    }
  };

  return (
    <section
      data-testid="workspaces-admin"
      className="flex flex-col gap-4 p-4"
    >
      <h2 className="text-lg font-semibold">Workspaces</h2>
      {error && <p className="text-xs text-red-600">{error}</p>}
      <form
        onSubmit={submit}
        data-testid="workspaces-create-form"
        className="flex items-end gap-2"
      >
        <label className="flex flex-col gap-1 text-xs">
          Name
          <Input
            value={name}
            onChange={(e) => setName(e.target.value)}
            placeholder="team-x"
            data-testid="workspaces-create-name"
          />
        </label>
        <label className="flex flex-col gap-1 text-xs">
          Plan
          <select
            data-testid="workspaces-create-plan"
            value={plan}
            onChange={(e) => setPlan(e.target.value)}
            className="rounded border px-2 py-1 text-xs"
          >
            <option value="free">free</option>
            <option value="team">team</option>
            <option value="pro">pro</option>
          </select>
        </label>
        <Button
          type="submit"
          data-testid="workspaces-create-submit"
          disabled={loading || !name.trim()}
        >
          {loading ? "Creating…" : "Create"}
        </Button>
      </form>
      <table data-testid="workspaces-table" className="w-full text-sm">
        <thead>
          <tr className="text-left">
            <th className="p-2">Name</th>
            <th className="p-2">Plan</th>
            <th className="p-2">Created</th>
          </tr>
        </thead>
        <tbody>
          {items.length === 0 ? (
            <tr data-testid="workspaces-empty">
              <td colSpan={3} className="p-4 text-center italic text-muted-foreground">
                No workspaces yet
              </td>
            </tr>
          ) : (
            items.map((w) => (
              <tr key={w.id} data-testid={`workspace-row-${w.id}`}>
                <td className="p-2">{w.name}</td>
                <td className="p-2">{w.plan}</td>
                <td className="p-2">
                  {new Date(w.created_at).toLocaleString()}
                </td>
              </tr>
            ))
          )}
        </tbody>
      </table>
    </section>
  );
}
```
