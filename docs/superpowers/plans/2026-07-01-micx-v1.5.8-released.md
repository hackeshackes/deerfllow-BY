```python
"""Comments HTTP endpoints.
Wires the in-memory CommentStore to FastAPI. Persistence is added in
v1.5.9+ via data/migrations/004_v1.5.8.sql.
"""
from __future__ import annotations

from datetime import datetime, timezone
from typing import Optional

from fastapi import APIRouter, Depends, HTTPException, Request
from pydantic import BaseModel, Field

from app.gateway.auth import require_user
from app.gateway.comments.models import Comment
from app.gateway.comments.service import InMemoryCommentStore


def _get_store(request: Request) -> InMemoryCommentStore:
    """Resolve the comment store from app.state (singleton; set in app.py)."""
    store = getattr(request.app.state, "comments_store", None)
    if store is None:
        # Fallback to a fresh in-memory store so the endpoint works even
        # before app.py wires app.state. Useful for tests + first boot.
        store = InMemoryCommentStore()
    return store


router = APIRouter(prefix="/api/threads", tags=["comments"])


class CreateCommentRequest(BaseModel):
    body: str = Field(..., min_length=1, max_length=4000)
    parent_id: Optional[str] = None
    mentions: list[str] = Field(default_factory=list)


class CommentResponse(BaseModel):
    id: str
    thread_id: str
    author: str
    body: str
    parent_id: Optional[str]
    mentions: list[str]
    created_at: datetime

    @classmethod
    def from_model(cls, c: Comment) -> "CommentResponse":
        return cls(
            id=c.id,
            thread_id=c.thread_id,
            author=c.author,
            body=c.body,
            parent_id=c.parent_id,
            mentions=list(c.mentions or []),
            created_at=c.created_at,
        )


@router.get("/{thread_id}/comments", response_model=list[CommentResponse])
async def list_comments(
    thread_id: str,
    request: Request,
    user: dict = Depends(require_user),
) -> list[CommentResponse]:
    store = _get_store(request)
    items = store.list_for_thread(thread_id)
    return [CommentResponse.from_model(c) for c in items]


@router.post(
    "/{thread_id}/comments",
    response_model=CommentResponse,
    status_code=201,
)
async def create_comment(
    thread_id: str,
    payload: CreateCommentRequest,
    request: Request,
    user: dict = Depends(require_user),
) -> CommentResponse:
    store = _get_store(request)
    comment = store.add(
        thread_id=thread_id,
        body=payload.body,
        author=user.get("email") or user.get("id", "unknown"),
        parent_id=payload.parent_id,
        mentions=payload.mentions,
    )
    return CommentResponse.from_model(comment)


@router.delete(
    "/{thread_id}/comments/{comment_id}",
    status_code=204,
)
async def delete_comment(
    thread_id: str,
    comment_id: str,
    request: Request,
    user: dict = Depends(require_user),
) -> None:
    store = _get_store(request)
    existing = store.get(comment_id)
    if existing is None or existing.thread_id != thread_id:
        raise HTTPException(status_code=404, detail="comment not found")
    if existing.author != user.get("email"):
        raise HTTPException(status_code=403, detail="cannot delete others' comments")
    store.delete(comment_id)
```

        json={"body": "hello world", "mentions": ["bob"]},

    assert r.status_code == 201
    body = r.json()
    assert body["body"] == "hello world"
    assert body["author"] == "alice@example.com"
    assert body["mentions"] == ["bob"]

    listing = comments_test_app.get("/api/threads/t-1/comments").json()

-- v1.5.8 multitenancy + comments schema
-- Idempotent — safe to run multiple times.

CREATE TABLE IF NOT EXISTS workspaces (
    id          TEXT PRIMARY KEY,
    name        TEXT NOT NULL,
    owner_email TEXT NOT NULL,
    plan        TEXT NOT NULL DEFAULT 'free',
    created_at  TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS workspace_memberships (
    workspace_id TEXT NOT NULL,
    user_email   TEXT NOT NULL,
    role         TEXT NOT NULL DEFAULT 'member',
    joined_at    TEXT NOT NULL,
    PRIMARY KEY (workspace_id, user_email),
    FOREIGN KEY (workspace_id) REFERENCES workspaces(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS comments (
    id         TEXT PRIMARY KEY,
    thread_id  TEXT NOT NULL,
    author     TEXT NOT NULL,
    body       TEXT NOT NULL,
    parent_id  TEXT,
    created_at TEXT NOT NULL
);
CREATE INDEX IF NOT EXISTS idx_comments_thread ON comments(thread_id);

CREATE TABLE IF NOT EXISTS resource_quotas (
    workspace_id TEXT NOT NULL,
    scope        TEXT NOT NULL,           -- 'tokens' | 'rpm' | etc.
    limit_value  INTEGER NOT NULL,
    period       TEXT NOT NULL DEFAULT 'monthly',
    updated_at   TEXT NOT NULL,
    PRIMARY KEY (workspace_id, scope, period)
);

**Step 1:** 新建 `ThreadComments.tsx`:

```tsx
"use client";

import { useEffect, useState } from "react";

import { MentionInput, type MentionSuggestion } from "./MentionInput";

interface ApiComment {
  id: string;
  thread_id: string;
  author: string;
  body: string;
  mentions?: string[];
  created_at: string;
}

interface ThreadCommentsProps {
  threadId: string;
  fetchUrl?: string;   // override-able for tests
  emptyText?: string;
}

export function ThreadComments({
  threadId,
  fetchUrl,
  emptyText = "No comments yet — be the first.",
}: ThreadCommentsProps) {
  const [comments, setComments] = useState<ApiComment[]>([]);
  const [draft, setDraft] = useState("");
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const url = fetchUrl ?? `/api/threads/${threadId}/comments`;

  const reload = async () => {
    try {
      const r = await fetch(url, { credentials: "include" });
      if (r.ok) setComments(await r.json());
    } catch (e) {
      setError(e instanceof Error ? e.message : String(e));
    }
  };

  useEffect(() => {
    void reload();
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [threadId]);

  const submit = async () => {
    if (!draft.trim()) return;
    setLoading(true);
    setError(null);
    try {
      const r = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        credentials: "include",
        body: JSON.stringify({ body: draft }),
      });
      if (!r.ok) throw new Error(`POST failed: ${r.status}`);
      setDraft("");
      await reload();
    } catch (e) {
      setError(e instanceof Error ? e.message : String(e));
    } finally {
      setLoading(false);
    }
  };

  return (
    <section data-testid="thread-comments" className="flex flex-col gap-3 p-3">
      <h3 className="text-sm font-medium">Comments</h3>
      {error && <p className="text-xs text-red-600">{error}</p>}
      <ol className="flex flex-col gap-2">
        {comments.length === 0 ? (
          <li className="text-muted-foreground text-xs italic">{emptyText}</li>
        ) : (
          comments.map((c) => (
            <li
              key={c.id}
              data-testid={`thread-comment-${c.id}`}
              className="rounded border p-2 text-sm"
            >
              <div className="text-muted-foreground text-xs">
                {c.author} · {new Date(c.created_at).toLocaleString()}
              </div>
              <p className="mt-1">{c.body}</p>
            </li>
          ))
        )}
      </ol>
      <MentionInput
        value={draft}
        onChange={setDraft}
        placeholder="Add a comment…"
        suggestions={[] as MentionSuggestion[]}
      />
      <button
        type="button"
        data-testid="thread-comment-submit"
        disabled={loading || !draft.trim()}
        onClick={() => void submit()}
        className="bg-primary text-primary-foreground rounded px-3 py-1 text-sm disabled:opacity-50"
      >
        {loading ? "Posting…" : "Post comment"}
      </button>
    </section>
  );
}
```

import { cleanup, fireEvent, render, screen, waitFor } from "@testing-library/react";
import { afterEach, beforeEach, describe, expect, it, vi } from "vitest";

import { ThreadComments } from "./ThreadComments";

const mockFetch = vi.fn();
beforeEach(() => {
  mockFetch.mockReset();
  // @ts-expect-error — global fetch for tests
  global.fetch = mockFetch;
});
afterEach(() => cleanup());

describe("ThreadComments", () => {
  it("renders empty state when there are no comments", async () => {
    mockFetch.mockResolvedValueOnce({
      ok: true,
      json: async () => [],
    });
    render(<ThreadComments threadId="t-1" />);
    expect(
      await screen.findByText(/No comments yet/i),
    ).toBeInTheDocument();
  });

  it("submits a comment and refreshes the list", async () => {
    mockFetch
      .mockResolvedValueOnce({ ok: true, json: async () => [] }) // initial fetch
      .mockResolvedValueOnce({ ok: true, json: async () => ({ id: "c1" }) }) // POST
      .mockResolvedValueOnce({
        ok: true,
        json: async () => [

            id: "c1",
            thread_id: "t-1",
            author: "alice",
            body: "hi",
            created_at: "2026-07-01T00:00:00Z",
          },
        ],

});
