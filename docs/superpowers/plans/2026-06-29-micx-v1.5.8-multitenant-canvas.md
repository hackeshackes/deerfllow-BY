- **后端线**: 多租户模型 + 配额 + 路由 + 看板 + 可观测性

```python
"""Tests for the multitenancy domain model."""
import pytest
from app.gateway.multitenancy.models import (
    Tenant, Workspace, Project, ResourceQuota, QuotaPeriod,
)


def test_workspace_belongs_to_tenant():
    t = Tenant(id="t-1", name="Acme")
    w = Workspace(id="w-1", name="Eng", tenant_id=t.id)
    assert w.tenant_id == t.id


def test_project_nests_in_workspace():
    t = Tenant(id="t-1", name="Acme")
    w = Workspace(id="w-1", name="Eng", tenant_id=t.id)
    p = Project(id="p-1", name="API", workspace_id=w.id)
    assert p.workspace_id == w.id


def test_quota_period_monthly():
    q = ResourceQuota(
        tenant_id="t-1",
        period=QuotaPeriod.MONTHLY,
        max_tokens=10_000_000,
        max_rpm=60,
    )
    assert q.period == QuotaPeriod.MONTHLY
    assert q.max_tokens == 10_000_000


def test_quota_rejects_negative():
    with pytest.raises(ValueError, match="max_tokens"):
        ResourceQuota(
            tenant_id="t-1",
            period=QuotaPeriod.MONTHLY,
            max_tokens=-1,
            max_rpm=60,
        )
```

```python
"""Tests for ModelRouter — cost/quality/speed strategy."""
import pytest
from app.gateway.multitenancy.router_service import (
    ModelRouter, RoutingStrategy,
)
from app.gateway.multitenancy.cost_calculator import ModelCost


def test_cost_strategy_picks_cheapest():
    costs = {
        "cheap": ModelCost(name="cheap", input_per_1k=0.001, output_per_1k=0.002),
        "expensive": ModelCost(name="expensive", input_per_1k=0.01, output_per_1k=0.02),
    }
    router = ModelRouter(strategy=RoutingStrategy.COST, costs=costs)
    assert router.pick() == "cheap"


def test_quality_strategy_picks_highest():
    """Quality strategy currently uses a static 'best' model."""
    costs = {"best": ModelCost(name="best", input_per_1k=0.01, output_per_1k=0.02)}
    router = ModelRouter(
        strategy=RoutingStrategy.QUALITY,
        costs=costs,
        quality_map={"best": 0.95, "old": 0.7},
    )
    assert router.pick() == "best"


def test_unknown_strategy_raises():
    from app.gateway.multitenancy.router_service import ModelRouter
    with pytest.raises(ValueError, match="unknown strategy"):
        ModelRouter(strategy="garbage", costs={})  # type: ignore[arg-type]
```
